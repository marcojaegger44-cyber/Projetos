def calcular_valor(preco, desconto):    # 🟢 Funções em verde
    """Calcula valor com desconto"""    # 🟤 Docstrings em marrom
    
    # ! Alerta importante               # ❗ Alerta em vermelho

        (desconto >= 0)):              # 🟡 Parênteses nível 2

    return 0                           # 🟣 Números em roxo
